Component.extend({ tag: "todos-list", view: `
    {{# each(todos) }}
  • {{ name }}
    • {{/ each }}
There are {{ completeCount }} completed todo(s). {{# is(completeCount, todos.length) }} You completed all the todos! {{ else }} Some todos remain. {{/ is }} `, ViewModel: { todos: { default: () => [ {complete: true, name: "Do the dishes."}, {complete: true, name: "Wash the car."}, {complete: false, name: "Learn CanJS."} ] }, get completeCount(){ return this.todos.filter({complete: true}).length }, completeAll(){ this.todos.forEach(function(todo){ todo.complete= true; }) } } });
